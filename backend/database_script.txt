-- 1. RECREACIÓN DE LA BASE DE DATOS
USE master;
GO

IF EXISTS (SELECT * FROM sys.databases WHERE name = 'horizon_db')
BEGIN
    ALTER DATABASE horizon_db SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
    DROP DATABASE horizon_db;
END
GO

CREATE DATABASE horizon_db;
GO

USE horizon_db;
GO

-- 2. ESTRUCTURA DE TABLAS

-- Tabla: Operadores (Socios estratégicos)
CREATE TABLE operadores (
    id_operador INT IDENTITY(1,1) PRIMARY KEY,
    contacto VARCHAR(100) NOT NULL,
    telefono VARCHAR(50),
    fecha_registro DATETIME DEFAULT GETDATE(),
    estado VARCHAR(20) DEFAULT 'Activo' -- Activo / Inactivo
);
GO

-- Tabla: Servicios (Experiencias turísticas)
CREATE TABLE servicios (
    id_servicio INT IDENTITY(1,1) PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    destino VARCHAR(100) NOT NULL,
    inversion DECIMAL(10, 2),
    capacidad INT,
    id_operador INT,
    estado VARCHAR(20) DEFAULT 'Activo', -- Activo / Inactivo
    CONSTRAINT FK_Servicio_Operador FOREIGN KEY (id_operador) 
        REFERENCES operadores(id_operador)
);
GO

-- Tabla: Reservas (Gestión de ventas)
CREATE TABLE reservas (
    id_reserva INT IDENTITY(1,1) PRIMARY KEY,
    cliente VARCHAR(100) NOT NULL,
    id_servicio INT,
    fecha_reserva DATETIME DEFAULT GETDATE(),
    estado_reserva VARCHAR(20) DEFAULT 'Pendiente', -- Confirmada / Pendiente / Cancelada
    estado_pago VARCHAR(20) DEFAULT 'Pendiente',    -- Pagado / Pendiente / No Pagado
    CONSTRAINT FK_Reserva_Servicio FOREIGN KEY (id_servicio) 
        REFERENCES servicios(id_servicio)
);
GO

-- 3. PROCEDIMIENTOS ALMACENADOS (Lógica de Negocio)

-- --- OPERADORES ---
CREATE PROCEDURE sp_get_operadores
AS
BEGIN
    SELECT id_operador, contacto, telefono, fecha_registro, estado 
    FROM operadores ORDER BY id_operador DESC;
END
GO

CREATE PROCEDURE sp_create_operador
    @contacto VARCHAR(100),
    @telefono VARCHAR(50),
    @estado VARCHAR(20)
AS
BEGIN
    INSERT INTO operadores (contacto, telefono, estado)
    VALUES (@contacto, @telefono, @estado);
END
GO

CREATE PROCEDURE sp_update_operador
    @id_operador INT,
    @contacto VARCHAR(100),
    @telefono VARCHAR(50),
    @estado VARCHAR(20)
AS
BEGIN
    UPDATE operadores SET 
        contacto = @contacto,
        telefono = @telefono,
        estado = @estado
    WHERE id_operador = @id_operador;
END
GO

CREATE PROCEDURE sp_delete_operador
    @id_operador INT
AS
BEGIN
    -- Borrado en cascada manual para integridad referencial
    DELETE FROM reservas WHERE id_servicio IN (SELECT id_servicio FROM servicios WHERE id_operador = @id_operador);
    DELETE FROM servicios WHERE id_operador = @id_operador;
    DELETE FROM operadores WHERE id_operador = @id_operador;
END
GO

-- --- SERVICIOS ---
CREATE PROCEDURE sp_get_servicios
AS
BEGIN
    SELECT s.*, o.contacto as nombre_operador 
    FROM servicios s
    LEFT JOIN operadores o ON s.id_operador = o.id_operador
    ORDER BY s.id_servicio DESC;
END
GO

CREATE PROCEDURE sp_create_servicio
    @nombre VARCHAR(100),
    @destino VARCHAR(100),
    @inversion DECIMAL(10, 2),
    @capacidad INT,
    @id_operador INT,
    @estado VARCHAR(20)
AS
BEGIN
    INSERT INTO servicios (nombre, destino, inversion, capacidad, id_operador, estado)
    VALUES (@nombre, @destino, @inversion, @capacidad, @id_operador, @estado);
END
GO

CREATE PROCEDURE sp_update_servicio
    @id_servicio INT,
    @nombre VARCHAR(100),
    @destino VARCHAR(100),
    @inversion DECIMAL(10, 2),
    @capacidad INT,
    @id_operador INT,
    @estado VARCHAR(20)
AS
BEGIN
    UPDATE servicios SET 
        nombre = @nombre,
        destino = @destino,
        inversion = @inversion,
        capacidad = @capacidad,
        id_operador = @id_operador,
        estado = @estado
    WHERE id_servicio = @id_servicio;
END
GO

CREATE PROCEDURE sp_delete_servicio
    @id_servicio INT
AS
BEGIN
    DELETE FROM reservas WHERE id_servicio = @id_servicio;
    DELETE FROM servicios WHERE id_servicio = @id_servicio;
END
GO

-- --- RESERVAS ---
CREATE PROCEDURE sp_get_reservas
AS
BEGIN
    SELECT r.*, s.nombre as nombre_servicio 
    FROM reservas r
    LEFT JOIN servicios s ON r.id_servicio = s.id_servicio
    ORDER BY r.id_reserva DESC;
END
GO

CREATE PROCEDURE sp_create_reserva
    @cliente VARCHAR(100),
    @id_servicio INT,
    @estado_reserva VARCHAR(20),
    @estado_pago VARCHAR(20)
AS
BEGIN
    INSERT INTO reservas (cliente, id_servicio, estado_reserva, estado_pago)
    VALUES (@cliente, @id_servicio, @estado_reserva, @estado_pago);
END
GO

CREATE PROCEDURE sp_update_reserva
    @id_reserva INT,
    @cliente VARCHAR(100),
    @id_servicio INT,
    @estado_reserva VARCHAR(20),
    @estado_pago VARCHAR(20)
AS
BEGIN
    UPDATE reservas SET 
        cliente = @cliente,
        id_servicio = @id_servicio,
        estado_reserva = @estado_reserva,
        estado_pago = @estado_pago
    WHERE id_reserva = @id_reserva;
END
GO

CREATE PROCEDURE sp_delete_reserva
    @id_reserva INT
AS
BEGIN
    DELETE FROM reservas WHERE id_reserva = @id_reserva;
END
GO
